41/* Depths of different drawing layers
42 *
43 * Format: <src><src><src><col><spr><spr><spr><spr>
44 *
45 * <src> : Indicates where the displayed pixel comes from
46 * <col> : Indicator bit for sb-collision detection
47 * <spr> : Sprite number
48 *
49 * Examples:
50 *
51 * 001 0 0000 : Border pixel
52 * 011 1 0000 : Foreground pixel
53 * 010 0 0000 : Sprite 0 (without sb-collision)
54 * 010 1 0101 : Sprite 6 (with sb-collision)
55 */
56#define DEPTH_BORDER 0b00100000 // In front of everything
57#define DEPTH_SPRITE_FG 0b01000000 // Behind border
58#define DEPTH_FG 0b01110000 // Behind sprite 1 layer
59#define DEPTH_SPRITE_BG 0b10000000 // Behind foreground
60#define DEPTH_BG 0b10100000 // Behind sprite 2 layer
61
62// Event flags
63#define VICUpdateIrqLine (1ULL << 0) // Sets or releases the IRQ line
64#define VICLpTransition (1ULL << 1) // Triggers a lightpen event
65#define VICUpdateFlipflops (1ULL << 2) // Updates the flipflop value pipeline
66#define VICUpdateRegisters (1ULL << 3) // Updates the register value pipeline
67#define VICUpdateBankAddr (1ULL << 4) // Updates the bank address
68#define VICSetDisplayState (1ULL << 5) // Flagged when control reg 1 changes
69#define VICClrSprSprCollReg (1ULL << 6) // Resets the sprite-sprite coll reg
70#define VICClrSprBgCollReg (1ULL << 7) // Resets the sprite-background coll reg
71
72#define VICClearanceMask \
73~((1ULL << 8) | \
74VICUpdateIrqLine | \
75VICLpTransition | \
76VICUpdateFlipflops | \
77VICUpdateRegisters | \
78VICUpdateBankAddr | \
79VICSetDisplayState | \
80VICClrSprSprCollReg | \
81VICClrSprBgCollReg);
